                if (window.location.pathname == '/users/zalerts.php' || window.location.pathname == '/zalerts.php') {

                    $('#zalertDialog').modal('hide')

            }).fetch()

        createForm.zalertCreateButton.removeAttribute('disabled')

    const validateQuery = (() => {
        let isQueryValid = false
        const zalertCreateButton = document.forms.zalertCreateForm.zalertCreateButton
        const zalertQueryInput = document.forms.zalertCreateForm.zalertQuery

        return () => {
            const newValueLength = [...zalertQueryInput.value].length
            const isQueryValidityStatusChanged = (() =>
                ((!isQueryValid && newValueLength >= 4) || (isQueryValid && newValueLength < 4)))()

            if (isQueryValidityStatusChanged) {
                zalertCreateButton.toggleAttribute('disabled')
                zalertCreateButton.classList.toggle('z-btn--preferable')
                isQueryValid = !isQueryValid
            }

    })()
